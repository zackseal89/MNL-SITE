  /* ── HERO IMAGE SLIDESHOW — ADDED ── */
  (function initSlideshow() {

    const SLIDES = document.querySelectorAll('.img-slide')
    const DOTS   = document.querySelectorAll('.dot')

    /* Caption text for each of the 4 slides in order */
    const CAPTIONS = [
      'MN Advocates LLP',
      "Nairobi's Legal Centre",
      'High Court of Kenya',
      'The MN Legal Boardroom'
    ]

    let current = 0
    let timer   = null

    function goTo(idx) {
      /* Deactivate current */
      SLIDES[current].classList.remove('active')
      SLIDES[current].classList.add('prev')
      DOTS[current].classList.remove('active')

      /* Clean up .prev after crossfade completes (1.4s + buffer) */
      const leaving = current
      setTimeout(function() {
        if (SLIDES[leaving]) SLIDES[leaving].classList.remove('prev')
      }, 1700)

      /* Activate next */
      current = idx
      SLIDES[current].classList.add('active')
      DOTS[current].classList.add('active')

      /* Update caption with fade */
      const capText = document.getElementById('cap-text')
      const capNum  = document.getElementById('cap-num')
      if (capText && capNum) {
        capText.style.opacity = '0'
        setTimeout(function() {
          capText.textContent = CAPTIONS[current]
          capNum.textContent  =
            String(current + 1).padStart(2, '0') +
            ' / ' +
            String(SLIDES.length).padStart(2, '0')
          capText.style.opacity = '1'
        }, 350)
      }
    }

    function nextSlide() {
      goTo((current + 1) % SLIDES.length)
    }

    function startTimer() {
      timer = setInterval(nextSlide, 5500)
    }

    function resetTimer() {
      clearInterval(timer)
      startTimer()
    }

    /* Dot click — jump to slide */
    DOTS.forEach(function(dot, i) {
      dot.addEventListener('click', function() {
        if (i === current) return
        goTo(i)
        resetTimer()
      })
      /* Extend cursor hover to dots */
      dot.addEventListener('mouseenter', function() {
        document.body.classList.add('cursor-hover')
      })
      dot.addEventListener('mouseleave', function() {
        document.body.classList.remove('cursor-hover')
      })
    })

    /* Pause slideshow on hover over image stage */
    var stage = document.getElementById('img-stage')
    if (stage) {
      stage.addEventListener('mouseenter', function() {
        clearInterval(timer)
      })
      stage.addEventListener('mouseleave', function() {
        startTimer()
      })
    }

    /* Keyboard accessibility — left/right arrows */
    document.addEventListener('keydown', function(e) {
      if (e.key === 'ArrowRight') { nextSlide(); resetTimer(); }
      if (e.key === 'ArrowLeft')  {
        goTo((current - 1 + SLIDES.length) % SLIDES.length)
        resetTimer()
      }
    })

    /* Start the slideshow */
    if (SLIDES.length > 0) startTimer()

  })()
  /* ── END SLIDESHOW ── */
